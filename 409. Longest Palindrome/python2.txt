class Solution(object):
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        l = []
        count = 0
        for c in s:
            if c not in l:
                l.append(c)
            else: 
                count += 1
                l.remove(c)
        return 2*count if (len(l) == 0) else 2*count + 1
