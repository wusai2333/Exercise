public class Solution {
    public String reverseString(String s) {
        char[] word = s.toCharArray();
        
        for(int i = 0, j = s.length() - 1; i < j; i++, j--){
            char tmp = word [i];
            word[i] = word[j];
            word[j] = tmp;
        }
        return new String(word);
    }
}
