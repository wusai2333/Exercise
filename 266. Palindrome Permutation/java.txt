public class Solution {
    public boolean canPermutePalindrome(String s) {
        int[] arr = new int[128];
        for(int i=0; i<s.length(); i++){
            int num = (int)s.charAt(i);
            arr[num]++;
        }
        int count = 0;
        for(int i=0; i<128; i++){
            if(arr[i]%2==1) count++;
            if(count>1) return false;
        }
        return true;
    }
}
