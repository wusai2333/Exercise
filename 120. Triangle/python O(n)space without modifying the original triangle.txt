class Solution(object):
    def minimumTotal(self, triangle):
        """
        :type triangle: List[List[int]]
        :rtype: int
        """
        minlen = triangle[-1]
        for i in range (len(triangle)-2, -1, -1):
            for j in range(i+1):
                minlen[j]  = min(minlen[j], minlen[j+1]) + triangle[i][j]
        return minlen[0]
