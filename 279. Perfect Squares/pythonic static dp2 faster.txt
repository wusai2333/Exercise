class Solution(object):
    _dp = [0]
    def numSquares(self, n):
        """
        :type n: int
        :rtype: int
        """
        dp = self._dp
        while len(dp) <= n:
            dp += min(dp[-i*i] for i in range(1, int(len(dp)**0.5+1))) + 1, #dp[-i*i] is dp[m - i*i], in which m is the current len(dp) 逗号可以把一个元素转化为元组类型
        return dp[n]
