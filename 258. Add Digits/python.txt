class Solution(object):
    def addDigits(self, num):
        """
        :type num: int
        :rtype: int
        """
        '''return num if not num else(num % 9 or 9);'''
        '''return (num - 1) % 9 + 1 if num else 0'''
        while(num >= 10):
            temp = 0
            while(num > 0):
                temp += num % 10
                num /= 10
            num = temp
        return num
