# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def swapPairs(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        dummy = ListNode(0)
        dummy.next = head
        curr = dummy
        while curr.next and curr.next.next :
            one = curr.next
            two = curr.next.next
            one.next =  two.next
            curr.next = two
            curr.next.next = one
            curr = curr.next.next
        return dummy.next
