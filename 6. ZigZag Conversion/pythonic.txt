class Solution(object):
    def convert(self, s, numRows):
        """
        :type s: str
        :type numRows: int
        :rtype: str
        """
        if numRows<=1 or numRows>=len(s): return s
        n = numRows -1
        step = n*2
        res = s[::step]
        for i in range(1,n):
            for v, w in itertools.izip_longest(s[i::step], s[step-i::step], fillvalue = ''):
                res += v+w
        return res + s[n::step]
