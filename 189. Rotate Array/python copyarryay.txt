class Solution(object):
    def rotate(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: void Do not return anything, modify nums in-place instead.
        """
        if len(nums) == 0 or k <= 0:
            return
        numscopy = nums[:]
        for i in range(len(nums)):
            nums[(i+k)%len(nums)] = numscopy[i]
