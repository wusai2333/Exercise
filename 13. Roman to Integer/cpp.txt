class Solution {
public:
    int romanToInt(string s) {
        unordered_map<char, int> dict = {{'M', 1000}, {'D',500}, {'C',100}, {'L', 50}, {'X', 10}, {'V', 5}, {'I', 1}};
        if(s.length() == 0) return 0;
        int sum = dict[s.back()];
        for (int i = 0; i < s.size() -1; i++) {
            if (dict[s[i]] < dict[s[i+1]]) 
                sum -= dict[s[i]];
            else 
                sum += dict[s[i]];
        }
        return sum;
    }
};
