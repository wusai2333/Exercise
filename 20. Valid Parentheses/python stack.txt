class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        stack = []
        for i in range(len(s)):
            if s[i] == '(' or s[i] == '[' or s[i] == '{':
                stack.append(s[i])
            elif stack and stack[-1] == '(' and s[i] == ')':
                stack.pop()
            elif stack and stack[-1] == '[' and s[i] == ']':
                stack.pop()
            elif stack and stack[-1] == '{' and s[i] == '}':
                stack.pop()
            else:
                return False
        return len(stack) == 0
