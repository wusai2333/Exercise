class Solution(object)
    def lengthOfLastWord(self, s)
        
        type s str
        rtype int
        
        length = 0
        l = len(s)-1
        while l  0 and s[l] == ' '
            l -= 1
        for i in range(l, -1, -1)
            if s[i] == ' '
                break
            length += 1

            
        return length
