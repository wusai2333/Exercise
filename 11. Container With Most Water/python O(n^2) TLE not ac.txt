class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        maxcontainer = 0
        n = len(height)
        for i in range(n):
                for j in range(i):
                    volume = (i-j)*min(height[i], height[j])
                    maxcontainer = max(maxcontainer, volume)
        return maxcontainer
