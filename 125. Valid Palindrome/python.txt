class Solution(object):
    def isPalindrome(self, s):
        """
        :type s: str
        :rtype: bool
        """
        l, r = 0, len(s) - 1
        while l < r:
            while l < r and s[l].isalnum():
                l += 1
            while r > l and s[r].isalnum():
                r -= 1
            if s[l].upper() != s[r]. upper():
                return False
            l += 1
            r -= 1
        return True
