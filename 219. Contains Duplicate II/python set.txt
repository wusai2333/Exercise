class Solution(object):
    def containsNearbyDuplicate(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: bool
        """
        set1 = set()
        for i, v in enumerate(nums):
            if i > k :
                set1.remove(nums[i-k-1])

            if nums[i] in set1:  
                return True
            set1.add(nums[i])
        return False
